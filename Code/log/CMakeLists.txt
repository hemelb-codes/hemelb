if(HEMELB_LOG_LEVEL MATCHES [Dd]ebug)
	set(logger LoggerDebug.cc)
elseif(HEMELB_LOG_LEVEL MATCHES [Ww]arning)
	set(logger  LoggerWarning.cc)
elseif(HEMELB_LOG_LEVEL MATCHES [Ii]nfo)
	set(logger LoggerInfo.cc)
else()
	message("Unknown log level ${HEMELB_LOG_LEVEL}, assuming info.")
	set(logger LoggerInfo.cc)
endif()
add_library(hemelb_log LoggerCommon.cc ${logger})
